# 62. implement BigInt addition

Luckily we have [BigInt](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/BigInt) in JavaScript so handle big numbers.

What if we need to do it by ourselves for older browsers?

You are asked to implement a string addition function, **with all non-negative integers in string**.

```js
add('999999999999999999', '1')
// '1000000000000000000'
```

Don't use BigInt directly, it is not our goal here.

## Solution

```js

function add(num1, num2) {
  let carry = 0;
  let result = [];
  // ğŸ’¡i: 2
  let i = num1.length - 1;
  // ğŸ’¡j: 1
  let j = num2.length - 1;
  
  // Round4: i = -1, j = -2, carray = 0 (go to return)
  while (i >= 0 || j >= 0 || carry) {
    // ğŸ’¡Round1: digit1 = 3
    // ğŸ’¡Round2: digit1 = 2
    // ğŸ’¡Round3: digit1 = 1
    const digit1 = i >= 0 ? parseInt(num1[i], 10) : 0;
    // ğŸ’¡Round1: digit2: 7
    // ğŸ’¡Round2: digit2 = 1
    // ğŸ’¡Round3: digit2 = 0
    const digit2 = j >= 0 ? parseInt(num2[j], 10) : 0;
	// ğŸ’¡Round1: sum = 3 + 7 + 0 (10)
    // ğŸ’¡Round2: sum = 2 + 1 + 1 (4)
    // ğŸ’¡Round3: sum = 1 + 0 + 0 (1)
    const sum = digit1 + digit2 + carry;
    // ğŸ’¡Round1: carray = 1
    // ğŸ’¡Round2: carray = 0
    // ğŸ’¡Round3: carray = 0
    carry = Math.floor(sum / 10);
    // ğŸ’¡Round1: result = [0]
    // ğŸ’¡Round2: result = [0, 4]
    // ğŸ’¡Round3: result = [0, 4, 1]
    result.push(sum % 10);
	
    // ğŸ’¡Round1: i = 1
    // ğŸ’¡Round2: i = 0
    // ğŸ’¡Round3: i = -1
    i--;
    // ğŸ’¡Round1: j = 0
    // ğŸ’¡Round2: j = -1
    // ğŸ’¡Round3: i = -2
    j--;
  }
  
  // ğŸ’¡[0, 4, 1]  => 140
  return result.reverse().join('');
}
```
